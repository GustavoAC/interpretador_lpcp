struct arvore {
  int num;
  ptr arvore sad;
  ptr arvore sae;
}

procedure inserir (ptr arvore tree, int numero) {
	arvore novo;
	novo.num = numero;
	if ($tree.num > numero) then
		if ($tree.sae == null) then
			$tree.sae = novo;
		else
			inserir(arvore.sae, numero);
		endif
	else
		if ($tree.sad == null) then
			$tree.sad = novo;
		else
			inserir(arvore.sad, numero);
		endif
	endif

}

function arvore_seq_valores ([int] seq): arvore{
	int contador_ex;
	contador_ex = 1;
	ptr arvore raizRef;
	arvore raiz;
	raizRef => raiz;
	raiz.num = seq[contador_ex];
	while (contador_ex < len(seq))
		
		inserir(raizRef, seq[contador_ex]);

	endwhile

	return raiz;

}

	if (tree.sad == null) then
